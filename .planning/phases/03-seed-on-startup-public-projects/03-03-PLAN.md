async def clean_projects_table() -> AsyncGenerator[None, None]:
    """Truncate `projects` and dependent tables before AND after each test.

                    await session.execute(text(stmt))
                except Exception:  # noqa: BLE001 — defensive cleanup
                    await session.rollback()
                    continue
            await session.commit()

    yield
    await _truncate()

    # re-reads from env.
    from ontokit.core.config import get_settings

    get_settings.cache_clear()
    yield git_dir
    get_settings.cache_clear()
