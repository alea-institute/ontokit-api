def _patch_fetch(mapping: dict[str, bytes | None]) -> Any:
    """Return an ``AsyncMock`` for ``_fetch_with_retry`` keyed by URL."""

    async def fake_fetch(url: str, *, max_bytes: int = seed_service.MAX_FETCH_BYTES) -> bytes | None:
        return mapping.get(url)

    return AsyncMock(side_effect=fake_fetch)


def _seed_settings(

async def test_seeds_two_projects_on_empty_db(
    tmp_path: Path, clean_projects_table: None
) -> None:
    """STARTUP-SEED-01/-04/-05: empty DB → 2 projects + 2 members + 2 repos + 2 index rows."""
    cfg = _seed_settings(tmp_path)
    fake_fetch = _patch_fetch(
        {
            cfg.seed_folio_url: _FOLIO_OWL,
            cfg.seed_canon_url: _CANON_TTL,
        }
    )

    with patch.object(seed_service, "_fetch_with_retry", new=fake_fetch):
        seeded = await seed_reference_projects(cfg)

    assert len(seeded) == 2
    assert {p.name for p in seeded} == {"Test FOLIO", "Test Canon"}

    async with async_session_maker() as session:
        rows = (await session.execute(select(Project))).scalars().all()
        assert len(rows) == 2
        for project in rows:
            assert project.is_public is True
            assert project.owner_id == cfg.seed_owner_id
            assert project.description and cfg.seed_folio_url in project.description or cfg.seed_canon_url in project.description

        members = (await session.execute(select(ProjectMember))).scalars().all()
        assert len(members) == 2
        assert all(m.role == "owner" for m in members)
        assert all(m.user_id == cfg.seed_owner_id for m in members)

        statuses = (
            await session.execute(select(OntologyIndexStatus))
        ).scalars().all()
        assert len(statuses) == 2
        for status in statuses:
            assert status.status == "ready"
            assert status.entity_count > 0

    git_service = get_git_service()
    for project in seeded:
        repo_path = Path(cfg.git_repos_base_path) / f"{project.id}.git"
        assert repo_path.exists(), f"bare repo missing for {project.name}"
        # Repository is initialized — at least one commit on main.
        repo = git_service.get_repository(project.id)
        commits = list(repo.list_commits(branch_name="main"))  # adapt to real method name
        assert len(commits) >= 1
