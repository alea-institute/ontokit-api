```python
"""Tests for the seed-on-startup service and its settings."""

from __future__ import annotations

import pytest

from ontokit.core.config import Settings


class TestSeedSettingsDefaults:
    """Default values and env overrides for the six SEED_* settings (D-12)."""

    def test_default_seed_sources_include_folio(self) -> None:
        """FOLIO defaults match D-12: name 'FOLIO', URL ends with FOLIO.owl."""
        s = Settings()
        assert s.seed_folio_name == "FOLIO"
        assert s.seed_folio_url.endswith("FOLIO.owl")
        assert s.seed_folio_url.startswith("https://")

    def test_default_seed_sources_include_canon(self) -> None:
        """Canon defaults match D-12: name 'Catholic Semantic Canon', .ttl URL."""
        s = Settings()
        assert s.seed_canon_name == "Catholic Semantic Canon"
        assert s.seed_canon_url.endswith("ontology-semantic-canon.ttl")
        assert s.seed_canon_url.startswith("https://")

    def test_seed_enabled_by_default(self) -> None:
        """SEED_REFERENCE_PROJECTS defaults to True (master toggle ON)."""
        s = Settings()
        assert s.seed_reference_projects is True

    def test_default_owner_id(self) -> None:
        """Default owner is the synthetic 'ontokit-system' user (D-03)."""
        s = Settings()
        assert s.seed_owner_id == "ontokit-system"

    def test_url_env_override_used(self, monkeypatch: pytest.MonkeyPatch) -> None:
        """STARTUP-SEED-09: overriding SEED_FOLIO_URL flows to Settings."""
        pinned = "https://example.test/pinned-folio.owl"
        monkeypatch.setenv("SEED_FOLIO_URL", pinned)
        s = Settings()
        assert s.seed_folio_url == pinned

    def test_disabled_via_settings(self, monkeypatch: pytest.MonkeyPatch) -> None:
        """STARTUP-SEED-07: SEED_REFERENCE_PROJECTS=false toggles the flag off."""
        monkeypatch.setenv("SEED_REFERENCE_PROJECTS", "false")
        s = Settings()
        assert s.seed_reference_projects is False
```
