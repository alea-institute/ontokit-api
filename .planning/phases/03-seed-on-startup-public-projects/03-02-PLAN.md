```python
"""Seed reference public projects (FOLIO, Catholic Semantic Canon) on first boot.

This module is the canonical "import OWL/Turtle from URL → Project + bare git
repo + Postgres ontology index" pipeline (D-02). It is invoked from the FastAPI
lifespan when ``settings.seed_reference_projects`` is true and may also be
called from the ``scripts/seed-project.py`` CLI introduced in PR #27.

Key invariants:

* Per-source isolation: each source uses its own ``AsyncSession`` so one
  source's failure cannot roll back another's commit.
* HTTP retry only: parse errors are deterministic and never retried (D-11).
* Idempotency by name: if a row exists in ``projects`` with the configured
  name, the source is skipped (D-06).
* Non-fatal: every exception is caught and logged; the caller (lifespan)
  always observes a clean return (D-10).
"""

from __future__ import annotations

import asyncio
import logging
from dataclasses import dataclass

import httpx
from rdflib import Graph
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from ontokit.core.config import Settings
from ontokit.core.config import settings as default_settings
from ontokit.core.database import async_session_maker
from ontokit.git import get_git_service
from ontokit.models.project import Project, ProjectMember
from ontokit.services.ontology_index import OntologyIndexService

logger = logging.getLogger(__name__)

# Ceiling on response size to mitigate parser-DoS (research § Security V10).
MAX_FETCH_BYTES: int = 50 * 1024 * 1024  # 50 MB

# Backoff schedule per D-09 ("3 attempts at 1s/2s/4s"). Sleep BEFORE attempts
# 2 and 3; total worst case ≈7 s wall clock (per A2 in research).
_RETRY_DELAYS: tuple[float, float, float] = (1.0, 2.0, 4.0)

# Hard-coded seed-commit author identity (per research assumption A5; not in
# v0.4.0 env surface).
_SEED_AUTHOR_NAME: str = "OntoKit Seed"
_SEED_AUTHOR_EMAIL: str = "seed@ontokit.local"


@dataclass(frozen=True)
class SeedSource:
    """A single reference ontology to seed."""

    name: str
    url: str
    rdflib_format: str  # "xml" | "turtle" | "n3" | "json-ld"
    filename: str  # filename committed inside the bare repo


def _format_for(url: str) -> tuple[str, str]:
    """Return ``(rdflib_format, filename)`` derived from the URL suffix.

    GitHub raw URLs return ``Content-Type: text/plain`` even for ``.owl``
    files, so URL-suffix detection is more reliable than the HTTP header
    (Pitfall 4 in research).
    """
    filename = url.rsplit("/", 1)[-1] or "ontology.owl"
    lower = filename.lower()
    if lower.endswith(".ttl"):
        return "turtle", filename
    if lower.endswith(".n3"):
        return "n3", filename
    if lower.endswith(".jsonld"):
        return "json-ld", filename
    # Default covers .owl, .rdf, .xml, and anything unknown.
    return "xml", filename


async def _fetch_with_retry(url: str, *, max_bytes: int = MAX_FETCH_BYTES) -> bytes | None:
    """Fetch ``url`` with 3 attempts at 1 s / 2 s / 4 s backoff.

    Returns the response body on success, or ``None`` after exhausting
    retries (logged WARNING). Returns ``None`` immediately if the
    ``Content-Length`` header exceeds ``max_bytes``.
    """
    last_err: Exception | None = None
    timeout = httpx.Timeout(10.0, connect=5.0)
    async with httpx.AsyncClient(timeout=timeout, follow_redirects=True) as client:
        for attempt, delay in enumerate(_RETRY_DELAYS, start=1):
            try:
                response = await client.get(url)
                response.raise_for_status()

                # Defensive content-length cap (research § Security V10).
                content_length = response.headers.get("content-length")
                if content_length is not None and int(content_length) > max_bytes:
                    logger.warning(
                        "Seed source %s rejected: Content-Length %s exceeds %d-byte cap",
                        url,
                        content_length,
                        max_bytes,
                    )
                    return None
                if len(response.content) > max_bytes:
                    logger.warning(
                        "Seed source %s rejected: body length %d exceeds %d-byte cap",
                        url,
                        len(response.content),
                        max_bytes,
                    )
                    return None
                return response.content
            except (httpx.HTTPError, httpx.TimeoutException) as exc:
                last_err = exc
                logger.info(
                    "Seed fetch attempt %d/%d for %s failed: %s",
                    attempt,
                    len(_RETRY_DELAYS),
                    url,
                    exc,
                )
                if attempt < len(_RETRY_DELAYS):
                    await asyncio.sleep(delay)
    logger.warning(
        "Failed to fetch seed source %s after %d attempts: %s",
        url,
        len(_RETRY_DELAYS),
        last_err,
    )
    return None


def _parse_graph(content: bytes, rdflib_format: str) -> Graph:
    """Synchronous rdflib parse — always invoked via ``asyncio.to_thread``."""
    graph = Graph()
    graph.parse(data=content, format=rdflib_format)
    return graph


async def _seed_one(source: SeedSource, owner_id: str) -> Project | None:
    """Seed a single source. Returns the new Project, or None on skip/failure."""
    async with async_session_maker() as session:
        # 1. Idempotency check (D-06).
        existing = await session.execute(
            select(Project).where(Project.name == source.name)
        )
        if existing.scalar_one_or_none() is not None:
            logger.info(
                "Reference project %r already exists, skipping", source.name
            )
            return None

        # 2. Fetch with retry (D-09); None means already-logged failure.
        content = await _fetch_with_retry(source.url)
        if content is None:
            return None

        # 3. Parse — no retry (D-11). Catch broadly per A3.
        try:
            graph = await asyncio.to_thread(
                _parse_graph, content, source.rdflib_format
            )
        except Exception as exc:
            logger.warning(
                "Failed to parse seed source %s as %s: %s",
                source.url,
                source.rdflib_format,
                exc,
            )
            return None

        # 4. Persist Project + owner ProjectMember in one transaction.
        project = Project(
            name=source.name,
            description=f"Reference ontology seeded from {source.url}",
            is_public=True,
            owner_id=owner_id,
            source_file_path=source.filename,
        )
        session.add(project)
        # Project.id is uuid4 Python-side default → already populated after add.
        session.add(
            ProjectMember(
                project_id=project.id,
                user_id=owner_id,
                role="owner",
            )
        )
        await session.commit()

        # 5. Bare git repo + initial commit (synchronous pygit2 → to_thread).
        git_service = get_git_service()
        commit_info = await asyncio.to_thread(
            git_service.initialize_repository,
            project.id,
            content,
            source.filename,
            _SEED_AUTHOR_NAME,
            _SEED_AUTHOR_EMAIL,
            source.name,
        )

        # 6. Postgres ontology index (commits internally on success).
        index_service = OntologyIndexService(session)
        await index_service.full_reindex(
            project.id, "main", graph, commit_info.hash
        )

        logger.info(
            "Seeded reference project %s (id=%s, %d triples)",
            source.name,
            project.id,
            len(graph),
        )
        return project


async def seed_reference_projects(cfg: Settings | None = None) -> list[Project]:
    """Seed the configured reference projects. Idempotent. Never raises.

    Returns the list of newly-created ``Project`` rows. Sources that were
    skipped (already present) or failed (network/parse) are absent from the
    returned list and logged at INFO/WARNING.
    """
    cfg = cfg or default_settings
    sources: list[SeedSource] = []
    for name, url in (
        (cfg.seed_folio_name, cfg.seed_folio_url),
        (cfg.seed_canon_name, cfg.seed_canon_url),
    ):
        rdflib_format, filename = _format_for(url)
        sources.append(
            SeedSource(
                name=name,
                url=url,
                rdflib_format=rdflib_format,
                filename=filename,
            )
        )

    seeded: list[Project] = []
    for source in sources:
        try:
            project = await _seed_one(source, cfg.seed_owner_id)
            if project is not None:
                seeded.append(project)
        except Exception:
            # D-10: never propagate. Lifespan must keep booting.
            logger.exception(
                "Unexpected error seeding %s — continuing", source.name
            )
    return seeded


class SeedService:
    """Thin OO wrapper for tests/DI parity with other services."""

    async def seed_reference_projects(
        self, cfg: Settings | None = None
    ) -> list[Project]:
        return await seed_reference_projects(cfg)


def get_seed_service() -> SeedService:
    """Factory for parity with other service singletons."""
    return SeedService()
```
